\begin{frame}[t]{Slide 4 \quad Methodology}
\vspace{2pt}
\begin{columns}[T]

  \begin{column}{0.48\textwidth}
    \textbf{\small RL Agents (Parts 2 \& 3)}
    \vspace{2pt}

    \begin{block}{\small DQN --- Part 2}
    \vspace{1pt}
    \footnotesize Off-policy Q-learning; $\varepsilon$-greedy; replay buffer\\
    Actions: \{Buy, Hold, Sell\}; 13-dim state vector
    \vspace{1pt}
    \end{block}

    \vspace{3pt}
    \begin{block}{\small REINFORCE + Baseline --- Part 3}
    \vspace{1pt}
    \footnotesize
    On-policy Monte Carlo PG; 2-layer MLP (128 units)\\
    $\nabla_\theta J = \mathbb{E}[\log\pi(a|s;\theta)\cdot(G_t - V(s))]$\\[2pt]
    \textbf{Reward-shaped trader archetypes:}\\
    \textcolor{accentblue}{Arb:} $r - 0.4\cdot\sigma_{24}\cdot f_\text{crypto}$\\
    \textcolor{cornellred}{Manip:} $r + 0.15\cdot\bar{V}\cdot\text{sign}(r_{4h})\cdot d$\\
    \textcolor{accentgreen}{Herder:} $r + 0.25\cdot\text{sign}(r_{24h})\cdot d$\\[2pt]
    \textbf{Explainability:} SHAP KernelExplainer on $\pi(\text{Buy}|s)$
    \vspace{1pt}
    \end{block}
  \end{column}

  \begin{column}{0.48\textwidth}
    \textbf{\small Bayesian Bandits (Part 4)}
    \vspace{2pt}

    \begin{block}{\small Setup}
    \vspace{1pt}
    \footnotesize
    $K=5$ arms; $Y_t=\mathbf{1}[\text{8h return}>0]$; Beta(3,3) prior; $T=500$
    \vspace{1pt}
    \end{block}

    \vspace{3pt}
    \begin{block}{\small Four Algorithms}
    \vspace{1pt}
    \footnotesize
    \begin{tabular}{>{\bfseries}p{1.8cm} p{4.0cm}}
    UCB1     & $\hat\mu_k + \sqrt{2\ln t / n_k}$ \\
    Bayes-UCB & $Q(1-1/t,\;\mathrm{Beta}(\alpha_k,\beta_k))$ \\
    Thompson  & $\hat\theta_k \sim \mathrm{Beta}(\alpha_k,\beta_k)$ \\
    Greedy    & $\arg\max_k\;\alpha_k/(\alpha_k+\beta_k)$ \\
    \end{tabular}\\[2pt]
    Info ratio: $\Gamma_t = [\mathbb{E}[\Delta_t]]^2 / I_t(\tilde\theta; A_t)$
    \vspace{1pt}
    \end{block}

    \vspace{3pt}
    \begin{block}{\small Market Simulation (Part 3)}
    \vspace{1pt}
    \footnotesize
    7 trader compositions; 500 test-set steps\\
    Signal: $S_t = w_\text{arb}d_\text{arb} + w_\text{man}d_\text{man} + w_\text{herd}d_\text{herd}$\\
    Metrics: crisis freq., stablecoin stability, efficiency
    \vspace{1pt}
    \end{block}
  \end{column}

\end{columns}

\begin{frame}[t]{Slide 5 \quad Key Findings}
\vspace{2pt}
\begin{columns}[T]

  \begin{column}{0.48\textwidth}
    \textbf{\small\textcolor{cornellred}{Trading Performance}}
    \begin{itemize}\setlength\itemsep{2pt}\small
      \item \red{REINFORCE-Manipulator:} \blu{$-$6\%} vs.
            $-$46\% buy-and-hold; preserved \blu{94\%} of capital
      \item DQN collapsed to constant HOLD\\
            (Q-values: Hold=0.0043, Buy=0.0041, Sell=0.0033)\\
            Stochastic RL adapts; deterministic RL does not
      \item Bayesian: Thompson Sampling regret \blu{9.19 $\pm$ 0.78};\\
            info ratio $\Gamma_T = \blu{0.149} \ll 0.5$ bound\\
            Optimal arm: \blu{BNB} ($\theta^*=0.519$)
    \end{itemize}

    \vspace{4pt}
    \textbf{\small\textcolor{cornellred}{Shapley Attribution}}
    \begin{itemize}\setlength\itemsep{2pt}\small
      \item \textcolor{accentblue}{\textbf{Arb:}} RSI + 24h return (mean reversion)
      \item \textcolor{cornellred}{\textbf{Manipulator:}} MA deviation + realized vol\\
            (pure microstructure --- zero trend signals)
      \item \textcolor{accentgreen}{\textbf{Herder:}} 7-day return dominant;\\
            blind to RSI, volume, contrarian signals
    \end{itemize}
  \end{column}

  \begin{column}{0.48\textwidth}
    \textbf{\small\textcolor{cornellred}{Systemic / Regulatory Findings}}
    \begin{itemize}\setlength\itemsep{2pt}\small
      \item Herder markets: \blu{6.75$\times$} more crisis events\\
            than arbitrageur markets (0.054 vs.\ 0.008)
      \item All-manipulator markets: \blu{0} measured crises\\
            but wealth extraction is \textit{silent}\\
            $\Rightarrow$ requires order-book surveillance
      \item Arbitrageur surge predicts USDT peg stress\\
            (lowest stability: 18.1 vs.\ 22.6 all-manipulator)
      \item \grn{Equal mix (33/33/34)} is the regulatory sweet spot:\\
            balanced crisis, stability, and efficiency
    \end{itemize}

    \vspace{4pt}
    \begin{alertblock}{\small Core Takeaway}
    \small
    Stochastic RL dominates deterministic Q-learning
    in volatile regimes. Manipulator activity is best
    detected via microstructure --- not price triggers.
    A diverse trader ecosystem minimizes systemic risk.
    \end{alertblock}
  \end{column}

\end{columns}
\end{frame}
